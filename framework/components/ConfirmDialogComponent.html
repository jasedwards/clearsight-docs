import { Component, Input, OnInit } from '@angular/core';
import { NgbActiveModal } from '@ng-bootstrap/ng-bootstrap';

@Component({
  selector: 'confirm-dialog',
  templateUrl: './confirm-dialog.component.html',
  styleUrls: ['./confirm-dialog.component.scss']
})
export class ConfirmDialogComponent implements OnInit {
  @Input() confirmText;
  @Input() confirmHeader;
  @Input() confirmHeaderIcon;
  @Input() confirmButtonText;
  @Input() cancelButtonText;
  @Input() confirmSubHeader;
  @Input() confirmClass = '';

  rootClass = '';

  constructor(private activeModal: NgbActiveModal) {}

  dismiss() {
    this.activeModal.dismiss(false);
  }

  close() {
    this.activeModal.close(true);
  }

  ngOnInit() {
    this.rootClass = `confirm-modal ${this.confirmClass}`;
  }
}

<div [class]="rootClass" role="alertdialog">

    </h4>
    <button class="close" type="button" csButtonTitle="STORMS_CLOSE" (click)="dismiss()">
      <span class="cs-icon-closed"></span>
    </button>
  </div>
  <div class="modal-body">
    <h5 *ngIf="confirmSubHeader">{{ confirmSubHeader }}</h5>
