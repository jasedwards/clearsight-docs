import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { DualListDragDrop } from './dual-list.directive';
import { IListItem } from '@clearsight-apps/models/list';
import { GlobalizationService } from '@clearsight-apps/global-services/src';

@Component({
  selector: 'cs-dual-list',
  templateUrl: './dual-list.component.html',
  styleUrls: ['./dual-list.component.scss']
})
export class DualListComponent implements OnInit {
  @Input() displayProperty: string;
  @Input() idProperty: string;
  @Input() allowMultiSelect: boolean;
  @Input() allowDragDrop: boolean;
  @Input() sourceItems: IListItem[];
  @Input() targetItems: IListItem[];
  @Input() sourceHeader: string;
  @Input() targetHeader: string;
  @Output() dropped = new EventEmitter<DualListDragDrop<IListItem>>();
  @Output() selectionChanged = new EventEmitter<IListItem[]>();
  @Output() listChanged = new EventEmitter<IListItem[]>();

  actionLabels: any;

  constructor(gService: GlobalizationService) {
    const addLabel = gService.translate('STORMS_ADD');
    const removeLabel = gService.translate('STORMS_REMOVE');
    const moveLabel = gService.translate('STORMS_MOVE');
    const allLabel = gService.translate('STORMS_ALL');
    const upLabel = gService.translate('STORMS_UP');
    const downLabel = gService.translate('STORMS_DOWN');
    this.actionLabels = {
      add: addLabel,
      remove: removeLabel,
      addAll: `${addLabel} ${allLabel}`,
      removeAll: `${removeLabel} ${allLabel}`,
      moveUp: `${moveLabel} ${upLabel}`,
      moveDown: `${moveLabel} ${downLabel}`
    };
  }

  ngOnInit() {}

  onDrop(ev) {
    this.dropped.emit(ev);
  }

  onSelectionChanged(items) {
    this.selectionChanged.emit(items);
  }

  onListChanged(items) {
    this.listChanged.emit(items);
  }
}
