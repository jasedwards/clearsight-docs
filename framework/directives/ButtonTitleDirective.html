import { AfterViewInit, Directive, ElementRef, Input, OnChanges, Renderer2, SimpleChanges } from '@angular/core';
import { GlobalizationService } from '@clearsight-apps/global-services';

@Directive({
  selector: '[csButtonTitle]'
})
export class ButtonTitleDirective implements AfterViewInit, OnChanges {
  @Input() csButtonTitle: string;
  @Input() isLink = false;
  @Input() insertText = false;

  constructor(private globalization: GlobalizationService, private el: ElementRef, private renderer: Renderer2) {}

  updateValues() {
    const typeAtt = this.isLink ? 'link' : 'button';
    const key = this.globalization.translate(this.csButtonTitle);
    this.renderer.setAttribute(this.el.nativeElement, 'aria-label', key);
    this.renderer.setAttribute(this.el.nativeElement, 'title', key);
    this.renderer.setAttribute(this.el.nativeElement, 'tabindex', '0');
    this.renderer.setAttribute(this.el.nativeElement, 'role', typeAtt);
    if (this.insertText && this.el.nativeElement) {
      if (this.el.nativeElement.childElementCount === 0) {
        const span_el = this.renderer.createElement('span');
        this.renderer.setProperty(span_el, 'innerText', key);
        this.renderer.appendChild(this.el.nativeElement, span_el);
      } else {
        this.el.nativeElement.children[0].innerText = key;
      }
    }

    if (!this.el.nativeElement.hasAttribute('type')) {
      this.renderer.setAttribute(this.el.nativeElement, 'type', typeAtt);
    }
    if (!this.el.nativeElement.id) {
      this.renderer.setAttribute(this.el.nativeElement, 'id', this.csButtonTitle);
    }
  }

  ngAfterViewInit() {
    this.updateValues();
  }

  ngOnChanges(changes: SimpleChanges) {
    if (changes && changes.csButtonTitle && !changes.csButtonTitle.firstChange) {
      this.updateValues();
    }
  }
}
