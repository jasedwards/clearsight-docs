import {
  AfterViewChecked,
  AfterViewInit,
  ContentChild,
  Directive,
  ElementRef,
  HostListener,
  NgZone,
  OnDestroy,
  Renderer2
} from '@angular/core';

@Directive({
  selector: '[csFreezeHeader]'
})
export class FreezeHeaderDirective implements AfterViewInit, AfterViewChecked, OnDestroy {
  @ContentChild('headerContainer', { static: false }) headerContainer: ElementRef;
  @ContentChild('headerTable', { static: false }) headerTable: ElementRef;
  @ContentChild('mainContainer', { static: false }) mainContainer: ElementRef;
  @ContentChild('mainTable', { static: false }) mainTable: ElementRef;

  private mainListener;
  private headerListener;
  private isLoaded = false;
  constructor(private el: ElementRef, private renderer: Renderer2, private zone: NgZone) {}

  private scrollHeaderTable() {
    this.zone.runOutsideAngular(() => {
      this.renderer.setProperty(
        this.headerContainer.nativeElement,
        'scrollLeft',
        this.mainContainer.nativeElement.scrollLeft
      );
    });
  }

  private scrollMainTable() {
    this.zone.runOutsideAngular(() => {
      this.renderer.setProperty(
        this.mainContainer.nativeElement,
        'scrollLeft',
        this.headerContainer.nativeElement.scrollLeft
      );
    });
  }

  private setColumnWidth() {
    this.renderer.setStyle(
      this.headerContainer.nativeElement,
      'minHeight',
      this.headerTable.nativeElement.offsetHeight + 'px'
    );
    this.renderer.setStyle(
      this.headerContainer.nativeElement,
      'right',
      this.mainContainer.nativeElement.offsetWidth - this.mainContainer.nativeElement.clientWidth + 'px'
    );
    const headerCells = this.headerTable.nativeElement.getElementsByTagName('th');
    const bodyCells: HTMLCollection = this.mainTable.nativeElement.getElementsByTagName('th');
    if (bodyCells && bodyCells.length) {
      for (let i = 0; i < bodyCells.length; i++) {
        if (headerCells[i]) {
          this.renderer.setStyle(headerCells[i], 'minWidth', (<HTMLElement>bodyCells[i]).offsetWidth + 'px');
        }
      }
    }
  }

  setMeasurements() {
    if (!this.isLoaded) {
      return;
    }
    const mainOffsetHeight = this.mainContainer.nativeElement.offsetHeight;
    const scrollBarHeight = mainOffsetHeight - this.mainContainer.nativeElement.clientHeight;
    const height =
      mainOffsetHeight + (this.el.nativeElement.offsetHeight - this.el.nativeElement.clientHeight) + scrollBarHeight;

    if (height <= 0) {
      return;
    }
    this.renderer.setStyle(this.el.nativeElement, 'height', height + 'px');
    this.setColumnWidth();
  }

  ngAfterViewInit() {
    if (!this.mainContainer || !this.headerContainer) {
      return;
    }
    this.zone.runOutsideAngular(() => {
      this.headerListener = this.renderer.listen(this.headerContainer.nativeElement, 'scroll', () => {
        this.scrollMainTable();
      });
      this.mainListener = this.renderer.listen(this.mainContainer.nativeElement, 'scroll', () => {
        this.scrollHeaderTable();
      });
      this.isLoaded = true;
      this.setMeasurements();
    });
  }

  @HostListener('window:resize', [])
  ngAfterViewChecked() {
    this.zone.runOutsideAngular(() => {
      setTimeout(() => {
        this.setMeasurements();
      });
    });
  }

  ngOnDestroy() {
    if (this.headerListener) {
      this.headerListener();
    }

    if (this.mainListener) {
      this.mainListener();
    }
  }
}
