import {
  AfterContentInit,
  ContentChild,
  Directive,
  ElementRef,
  EventEmitter,
  HostListener,
  Input,
  NgZone,
  OnDestroy,
  OnInit,
  Output,
  Renderer2
} from '@angular/core';
import { Subject, Subscription } from 'rxjs';
import { distinctUntilChanged, map } from 'rxjs/operators';
import { hasValue } from '@clearsight-apps/common-functions/has-value';

@Directive({
  selector: '[csResizePanel]'
})
export class ResizePanelDirective implements AfterContentInit, OnInit, OnDestroy {
  @Input() minWidth = 40;
  @Input() maxWidth: number;
  @Input() asPercent = false;
  @Input() rightSide = false;
  @Input() grabberClass: string;

  @Output() draggingChanged = new EventEmitter<boolean>();
  @ContentChild('resizable', { static: false }) sizableElement: ElementRef;
  isDragging = false;
  private sub = new Subject<number>();
  private mouseMove = new Subject<number>();
  private subscription: Subscription;
  private mouseMoveSubscription: Subscription;
  private mouseDown;
  private div;
  private canImplement = true;
  private isLoaded = false;

  @Input()
  set csResizePanel(val: boolean) {
    if (!hasValue(val)) {
      val = true;
    }
    if (val === this.canImplement) {
      return;
    }
    this.canImplement = val;

    if (!this.isLoaded) {
      return;
    }

    if (val) {
      this.setUp();
    } else {
      this.tearDown();
    }
  }

  static getPos(el: ElementRef, rightSide = false): number {
    const native: any = el.nativeElement;
    const boundingRec = native.getBoundingClientRect();
    return rightSide
      ? boundingRec.right + (window.pageXOffset || document.documentElement.scrollLeft)
      : boundingRec.left + (window.pageXOffset || document.documentElement.scrollLeft);
  }

  constructor(private renderer: Renderer2, private zone: NgZone, private el: ElementRef) {}

  private calcWidth(width: number): number {
    const minWidth = this.asPercent ? (this.minWidth / 100) * this.el.nativeElement.offsetWidth : this.minWidth;

    if (this.maxWidth) {
      const maxWidth = this.asPercent ? (this.maxWidth / 100) * this.el.nativeElement.offsetWidth : this.maxWidth;
      return width < minWidth ? minWidth : width > maxWidth ? maxWidth : width;
    }
    return width < minWidth ? minWidth : width;
  }

  private changeDragging(isDragging) {
    this.isDragging = isDragging;
    this.zone.run(() => {
      this.draggingChanged.emit(isDragging);
    });
  }

  private setUp() {
    const element = this.sizableElement.nativeElement;
    if (!this.div) {
      this.div = this.renderer.createElement('div');
      this.renderer.addClass(this.div, 'grabber');
      if (this.grabberClass) {
        this.renderer.addClass(this.div, this.grabberClass);
      }
    }

    // this.renderer.insertBefore(this.rightSide ? this.sizableElement.nativeElement : element.parentNode, this.div, this.rightSide ? element.previousSibling : element.nextSibling);
    if (this.rightSide) {
      this.renderer.insertBefore(element.parentNode, this.div, element);
    } else {
      this.renderer.insertBefore(element.parentNode, this.div, element.nextSibling);
    }

    this.subscription = this.sub.subscribe((val: number) => {
      const x = ResizePanelDirective.getPos(this.sizableElement, this.rightSide);
      if ((this.rightSide && val >= x) || (!this.rightSide && val <= x)) {
        this.changeDragging(false);
        return;
      }
      const width = this.rightSide ? x - val : val - x;
      this.zone.runOutsideAngular(() => {
        this.renderer.setStyle(element, 'width', this.calcWidth(width) + 'px');
      });
    });
    this.mouseDown = this.renderer.listen(this.div, 'mousedown', () => {
      this.changeDragging(true);
    });
  }

  private tearDown() {
    if (this.mouseDown) {
      this.mouseDown();
      this.mouseDown = null;
      this.changeDragging(false);
    }

    if (this.subscription) {
      this.subscription.unsubscribe();
      this.subscription = null;
    }

    if (this.div) {
      this.renderer.removeChild(this.sizableElement.nativeElement.parentNode, this.div);
    }
  }

  @HostListener('mouseup')
  mouseUp() {
    this.changeDragging(false);
  }

  @HostListener('mousemove', ['$event'])
  onMouseMove(event: MouseEvent): boolean {
    if (this.isDragging) {
      this.mouseMove.next(event.pageX);
    }
    return false;
  }

  ngAfterContentInit() {
    this.isLoaded = true;
    if (this.canImplement) {
      this.setUp();
    }
  }

  ngOnInit() {
    this.mouseMoveSubscription = this.mouseMove
      .pipe(
        map(v => v),
        distinctUntilChanged()
      )
      .subscribe(val => {
        this.sub.next(val);
      });
  }

  ngOnDestroy() {
    if (this.subscription) {
      this.subscription.unsubscribe();
    }

    if (this.mouseMoveSubscription) {
      this.mouseMoveSubscription.unsubscribe();
    }

    if (this.mouseDown) {
      this.mouseDown();
    }
  }
}
