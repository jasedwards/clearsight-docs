import { Directive, forwardRef, Input } from '@angular/core';
import { SelectDirective } from '@clearsight-apps/shared/framework/accessibility';

export enum ListAction {
  Add = 'Add',
  Remove = 'Remove',
  AddAll = 'AddAll',
  RemoveAll = 'RemoveAll',
  MoveUp = 'MoveUp',
  MoveDown = 'MoveDown'
}

@Directive({
  selector: '[csListAction]'
})
export class ListActionDirective extends SelectDirective {
  @Input('csListAction') type: ListAction = ListAction.Add;
  onSelect() {
    this.csSelect.emit(this.type);
  }
}
