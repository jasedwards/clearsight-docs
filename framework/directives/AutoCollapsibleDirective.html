import {
  Directive,
  ElementRef,
  AfterViewInit,
  HostListener,
  NgZone,
  Renderer2,
  Input,
  OnDestroy,
  ComponentFactoryResolver,
  ViewChild,
  ViewContainerRef,
  ComponentRef,
  Output,
  EventEmitter,
  AfterViewChecked
} from '@angular/core';
import { Observable, Observer, fromEvent } from 'rxjs';
import { DynamicMenuComponent } from './dynamic-menu/dynamic-menu.component';
import { debounceTime, distinctUntilChanged, map } from 'rxjs/operators';
import { Subscription } from 'rxjs';
import { GetStyles } from '@clearsight-apps/common-functions/get-styles';
import { GetNumberStyle } from '@clearsight-apps/common-functions/get-number-style';

@Directive({
  selector: '[csAutoCollapsible]',
  exportAs: 'csAutoCollapsible'
})
export class AutoCollapsibleDirective implements AfterViewInit, AfterViewChecked, OnDestroy {
  private childElements = [];
  private hiddenChildElements = [];
  private toggleButton: ComponentRef<DynamicMenuComponent>;
  private resizeSub: Subscription;
  private menuToggleSub: Subscription;
  private isCollapsed = false;
  private calculatedContentWidth = 0;
  private containerWidth = 0;
  private refreshObserver: Observer<any>;
  private refreshSub: Subscription;
  readonly offset = 30;

  private container: HTMLElement;

  @Input() addMenuSeparator: boolean;
  @Input()
  get autoCloseMenu() {
    return this.toggleButton ? this.toggleButton.instance.autoClose : true;
  }
  set autoCloseMenu(val) {
    if (this.toggleButton) {
      this.toggleButton.instance.autoClose = val;
    }
  }
  @Output() onCollapse: EventEmitter<boolean> = new EventEmitter();
  @Output() onMenuToggle: EventEmitter<boolean> = new EventEmitter();

  constructor(
    private renderer: Renderer2,
    private el: ElementRef,
    private zone: NgZone,
    private componentFactoryResolver: ComponentFactoryResolver,
    private viewContainerRef: ViewContainerRef
  ) {
    this.toggleButton = this.createToggleButton();

    this.refreshSub = new Observable(observer => (this.refreshObserver = observer))
      .pipe(debounceTime(200))
      .subscribe(() => {
        this.refresh();
      });
  }

  ngAfterViewInit() {
    this.initialize();
  }

  ngAfterViewChecked(): void {
    this.zone.runOutsideAngular(() => {
      this.refreshObserver.next(true);
    });
  }

  ngOnDestroy(): void {
    this.childElements = this.hiddenChildElements = null;
    this.toggleButton.destroy();
    this.toggleButton = null;

    if (this.resizeSub) {
      this.resizeSub.unsubscribe();
      this.resizeSub = null;
    }
    if (this.menuToggleSub) {
      this.menuToggleSub.unsubscribe();
      this.menuToggleSub = null;
    }
    if (this.refreshSub) {
      this.refreshSub.unsubscribe();
      this.refreshSub = null;
    }
    this.refreshObserver.complete();
    this.refreshObserver = null;
  }

  openMenu() {
    return this.toggleButton.instance.open();
  }
  closeMenu() {
    return this.toggleButton.instance.close();
  }
  toggleMenu() {
    return this.toggleButton.instance.toggle();
  }

  private createToggleButton() {
    const componentFactory = this.componentFactoryResolver.resolveComponentFactory(DynamicMenuComponent);
    const componentRef = this.viewContainerRef.createComponent(componentFactory);

    // this.renderer.setStyle(componentRef.location.nativeElement, 'order', '1');
    // this.renderer.setStyle(componentRef.location.nativeElement, 'justify-self', 'flex-end');
    // this.renderer.setStyle(componentRef.location.nativeElement, 'align-self', 'flex-start');
    this.menuToggleSub = componentRef.instance.onMenuToggle.subscribe(isOpen => {
      this.onMenuToggle.emit(isOpen);
    });
    return componentRef;
  }

  private initialize() {
    //this.renderer.setStyle(this.el.nativeElement, 'opacity', '0');
    this.container = this.renderer.createElement('div');

    this.renderer.setStyle(this.container, 'order', '0');
    this.renderer.setStyle(this.container, 'flex-grow', '1');
    this.renderer.addClass(this.container, 'auto-collapse-container');
    this.renderer.setStyle(this.el.nativeElement, 'position', 'relative');
    this.renderer.setStyle(this.el.nativeElement, 'padding', '0px');
    this.renderer.setStyle(this.el.nativeElement, 'display', 'flex');
    this.renderer.setStyle(this.el.nativeElement, 'flex-wrap', 'nowrap');
    this.hiddenChildElements.length = 0;
    this.childElements.length = 0;

    for (let index = 0; index < this.el.nativeElement.children.length; index++) {
      const el = this.el.nativeElement.children[index];
      this.childElements.push(el);
    }

    for (const el of this.childElements) {
      this.renderer.removeChild(this.el.nativeElement, el);
      this.renderer.appendChild(this.container, el);
    }
    this.renderer.appendChild(this.el.nativeElement, this.container);
    // this.renderer.removeChild(this.el.nativeElement.parentElement, this.toggleButton.location.nativeElement);
    // this.renderer.appendChild(this.el.nativeElement, this.toggleButton.location.nativeElement);

    if (this.container.querySelector('cs-dynamic-menu')) {
      this.renderer.removeChild(this.container, this.toggleButton.location.nativeElement);
    }
    this.renderer.appendChild(this.container, this.toggleButton.location.nativeElement);

    this.resizeSub = fromEvent(window, 'resize')
      .pipe(
        debounceTime(200),
        map(() => window.innerWidth),
        distinctUntilChanged()
      )
      .subscribe(() => {
        this.zone.runOutsideAngular(() => {
          this.setChildrenVisibility();
        });
      });
  }

  refresh() {
    if (!this.childElements) {
      return;
    }

    const contentWidth = this.calculateWidth(this.childElements);
    // If there is no major change in content width and container width then do not refresh.

    if (
      Math.abs(contentWidth - this.calculatedContentWidth) <= 10 &&
      Math.abs(this.containerWidth - this.el.nativeElement.offsetWidth) <= 10
    ) {
      return;
    }

    this.containerWidth = this.el.nativeElement.offsetWidth;

    this.calculatedContentWidth = contentWidth;
    this.zone.runOutsideAngular(() => {
      setTimeout(() => {
        this.setChildrenVisibility();
      });
    });
  }

  private setChildrenVisibility() {
    // Reinital container with making all children visible
    if (this.hiddenChildElements.length > 0) {
      this.childElements.map(el => {
        this.renderer.removeChild(this.container, el);
        this.renderer.appendChild(this.container, el);
      });
    }

    // Get available container width with reserved space for toggle button.  need to account for border that will be added to dd button

    const availableWidth = this.containerWidth - (this.toggleButton.location.nativeElement.offsetWidth + this.offset);

    let calcWidth = 0;
    this.hiddenChildElements = [];

    // Find chid elements which can  be hidden.
    const nonHidableChildElements = [];
    const hidableChildElements = [];
    this.childElements.map(child => {
      if (child.hasAttribute('disableHideOnCollapse')) {
        nonHidableChildElements.push(child);
      } else {
        hidableChildElements.push(child);
      }
    });

    // Container should have at least one child which is not hidden.
    if (nonHidableChildElements.length === 0 && hidableChildElements.length > 0) {
      nonHidableChildElements.push(hidableChildElements[0]);
      hidableChildElements.shift();
    }

    calcWidth = this.calculateWidth(nonHidableChildElements);

    // Iterate over child elements to determin which elements need to be hidden.
    for (const childEl of hidableChildElements) {
      const styles = GetStyles(childEl);

      const margin = GetNumberStyle(styles, 'marginRight') + GetNumberStyle(styles, 'marginLeft');
      calcWidth += childEl.offsetWidth + margin;

      if (calcWidth >= availableWidth) {
        this.hiddenChildElements.push(childEl);
        this.renderer.removeChild(this.container, childEl);
      }
    }

    this.renderer.removeChild(this.container, this.toggleButton.location.nativeElement);
    // If there are hidden child elements then add toggle button. Else remove it.
    if (this.hiddenChildElements.length > 0) {
      this.renderer.appendChild(this.container, this.toggleButton.location.nativeElement);
      this.renderer.setStyle(this.toggleButton.location.nativeElement, 'display', 'inline-block');
      // Set hidden child elements to be displayed in the menu.
      this.toggleButton.instance.addSeparator = this.addMenuSeparator;
      this.toggleButton.instance.Items = this.hiddenChildElements;

      if (!this.isCollapsed) {
        this.isCollapsed = true;
        this.onCollapse.emit(true);
      }
    } else {
      this.renderer.setStyle(this.toggleButton.location.nativeElement, 'display', 'none');

      if (this.isCollapsed) {
        this.isCollapsed = false;
        this.onCollapse.emit(false);
      }
    }
  }

  private calculateWidth(contentElements) {
    let contentWidth = 0;
    contentElements.map(item => {
      const styles = GetStyles(item);
      const margin = GetNumberStyle(styles, 'marginRight') + GetNumberStyle(styles, 'marginLeft');
      contentWidth += item.offsetWidth + margin;
    });

    return contentWidth;
  }
}
