import {
  Directive,
  EventEmitter,
  OnDestroy,
  Input,
  Output,
  QueryList,
  AfterContentInit,
  ContentChildren
} from '@angular/core';
import { Subscription, merge } from 'rxjs';
import { INavigationItem } from '@clearsight-apps/models/common/navigation-item';
import { NavDirection } from '@clearsight-apps/models/common/nav-direction';
import { NavigatorControlDirective } from '@clearsight-apps/shared/framework/accessibility/src/lib/item-list-navigator/navigator-control.directive';

@Directive({
  selector: '[csItemListNavigator]'
})
export class ItemListNavigatorDirective implements AfterContentInit, OnDestroy {
  @Input() items: Array<INavigationItem>;
  @Output() itemSelected = new EventEmitter<INavigationItem>();
  @ContentChildren(NavigatorControlDirective) navControls: QueryList<NavigatorControlDirective>;
  private sub: Subscription;

  constructor() {}

  ngAfterContentInit(): void {
    if (!this.items || this.items.length === 0 || !this.navControls || this.navControls.length === 0) {
      return;
    }

    this.sub = merge<NavDirection>(...this.navControls.map(ctrl => ctrl.csSelect)).subscribe(dir =>
      this.selectItem(dir)
    );
  }

  selectItem(direction: NavDirection) {
    let index = this.items.findIndex(item => item.Selected);
    this.items.forEach(item => (item.Selected = false));
    index = index < 0 ? 0 : index;
    switch (direction) {
      case NavDirection.Last: {
        index = this.items.length - 1;
        break;
      }
      case NavDirection.Prev: {
        index = index > 0 ? index - 1 : this.items.length - 1;
        break;
      }
      case NavDirection.Next: {
        index = index < this.items.length - 1 ? index + 1 : 0;
        break;
      }
      default:
        index = 0;
    }
    this.items[index].Selected = true;
    this.itemSelected.emit(this.items[index]);
  }

  ngOnDestroy(): void {
    if (this.sub) {
      this.sub.unsubscribe();
      this.sub = null;
    }
  }
}
