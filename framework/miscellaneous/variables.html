libs/shared/framework/browser/src/lib/scroll-to.const.ts

ScrollTo
Default value : (parent: HTMLElement, child: HTMLElement, padding = 30): void => { const parentTop = parent.getBoundingClientRect().top; const parentHeight = parent.clientHeight; const childTop = child.getBoundingClientRect().top; const childHeight = child.clientHeight; let scrollTop; if (childTop - padding < parentTop) { scrollTop = childTop - (childHeight + padding) + parent.scrollTop - parentTop; } else { scrollTop = childTop + parent.scrollTop - parentTop - (parentHeight - (childHeight + padding)); } parent.scrollTop = scrollTop; }
