import { Injectable } from '@angular/core';
import { NgbModal, NgbModalRef } from '@ng-bootstrap/ng-bootstrap';
import { ConfirmDialogComponent } from './confirm-dialog.component';
import { GlobalizationService } from '@clearsight-apps/global-services';

export interface IConfirmDialogService {
  open(any): NgbModalRef;
}

@Injectable({
  providedIn: 'root'
})
export class ConfirmDialogService implements IConfirmDialogService {
  constructor(protected modal: NgbModal, protected globalization: GlobalizationService) {}

  open({
    confirmText = 'STORMS_CONFIRM_TEXT',
    confirmHeader = 'STORMS_CONFIRM',
    confirmHeaderIcon = '',
    confirmButtonText = 'STORMS_OK',
    cancelButtonText = 'STORMS_CANCEL',
    confirmSubHeader = '',
    confirmClass = ''
  }: {
    confirmText?: string;
    confirmHeader?: string;
    confirmHeaderIcon?: string;
    confirmButtonText?: string;
    cancelButtonText?: string;
    confirmSubHeader?: string;
    confirmClass?: string;
  } = {}): NgbModalRef {
    const result = this.modal.open(ConfirmDialogComponent, { backdrop: 'static' });
    result.componentInstance.confirmText = this.globalization.translate(confirmText);
    result.componentInstance.confirmHeader = this.globalization.translate(confirmHeader);
    result.componentInstance.confirmHeaderIcon = this.globalization.translate(confirmHeaderIcon);
    result.componentInstance.confirmButtonText = this.globalization.translate(confirmButtonText);
    result.componentInstance.confirmSubHeader = this.globalization.translate(confirmSubHeader);
    result.componentInstance.confirmClass = this.globalization.translate(confirmClass);
    result.componentInstance.cancelButtonText = this.globalization.translate(cancelButtonText);
    return result;
  }
}
