import {
  Directive,
  ContentChildren,
  AfterContentInit,
  OnDestroy,
  ContentChild,
  Input,
  EventEmitter,
  Output
} from '@angular/core';
import { Subscription, merge } from 'rxjs';
import { SourceListDirective, TargetListDirective } from './list-type.directive';
import { IListItem, IListComponent } from '@clearsight-apps/models/list';
import { ListActionDirective, ListAction } from './list-action.directive';
import { CdkDragDrop } from '@angular/cdk/drag-drop';

export interface DualListDragDrop<T extends IListItem> extends CdkDragDrop<T[]> {
  droppedOnSource: boolean;
}

export interface DualListSelection {
  action: ListAction;
}

@Directive({
  selector: '[csDualList]'
})
export class DualListDirective implements AfterContentInit, OnDestroy {
  @Output() dropped = new EventEmitter<DualListDragDrop<IListItem>>();
  @Output() selectionChanged = new EventEmitter<{ SourceList: IListItem[]; TargetList: IListItem[] }>();
  @Output() listChanged = new EventEmitter<{ SourceList: IListItem[]; TargetList: IListItem[] }>();

  @ContentChildren(ListActionDirective) selectors: ListActionDirective[];
  @ContentChild(SourceListDirective, { static: false }) srcListDir: SourceListDirective;
  @ContentChild(TargetListDirective, { static: false }) targetListDir: TargetListDirective;

  private sourceList: IListComponent<IListItem>;
  private targetList: IListComponent<IListItem>;
  private sub = new Subscription();

  constructor() {}

  ngAfterContentInit(): void {
    if (!this.srcListDir || !this.srcListDir.hostComponent) {
      throw new Error('Source list component not found');
    }

    if (!this.targetListDir || !this.targetListDir.hostComponent) {
      throw new Error('Target list component not found');
    }

    this.sourceList = this.srcListDir.hostComponent;
    this.targetList = this.targetListDir.hostComponent;
    if (this.sourceList.dropped && this.targetList.dropped) {
      this.sub.add(this.sourceList.dropped.subscribe(event => this.dragdropItem(event, true)));
      this.sub.add(this.targetList.dropped.subscribe(event => this.dragdropItem(event, false)));
    }
    this.sub.add(
      this.targetList.listChanged.subscribe(() =>
        this.listChanged.emit({
          SourceList: [...this.sourceList.itemsList],
          TargetList: [...this.targetList.itemsList]
        })
      )
    );
    this.sub.add(
      this.sourceList.listChanged.subscribe(() =>
        this.listChanged.emit({
          SourceList: [...this.sourceList.itemsList],
          TargetList: [...this.targetList.itemsList]
        })
      )
    );

    this.sub.add(
      merge<ListAction>(...this.selectors.map(ctrl => ctrl.csSelect)).subscribe(action => this.performAction(action))
    );
  }

  private performAction(type: ListAction) {
    switch (type) {
      case ListAction.Add:
        return this.moveItems(this.sourceList, this.targetList);
      case ListAction.AddAll:
        return this.moveItems(this.sourceList, this.targetList, true);
      case ListAction.Remove:
        return this.moveItems(this.targetList, this.sourceList);
      case ListAction.RemoveAll:
        return this.moveItems(this.targetList, this.sourceList, true);
      case ListAction.MoveUp:
        return this.reorderItems(true);
      case ListAction.MoveDown:
        return this.reorderItems();
      default:
        console.log(`Action to be performed: ${type}`);
    }
  }

  private moveItems(srcList: IListComponent<IListItem>, trgList: IListComponent<IListItem>, all?: boolean) {
    if (!srcList.itemsList || !srcList.itemsList.length) {
      return;
    }
    let selectedItems = [];
    if (!all) {
      srcList.itemsList = srcList.itemsList.filter(item => {
        if (item.Selected) {
          selectedItems.push(item);
        }
        return !item.Selected;
      });
    } else {
      selectedItems = srcList.itemsList.filter(item => !item.DisableSelection);
      srcList.itemsList = srcList.itemsList.filter(item => item.DisableSelection);
    }

    if (!selectedItems || !selectedItems.length) {
      return;
    }
    trgList.itemsList = trgList.itemsList || [];
    trgList.itemsList = [...trgList.itemsList, ...selectedItems];
    trgList.itemsList.forEach(item => (item.Selected = false));

    //this.selectionChanged.emit([...this.targetList.itemsList]);
    this.selectionChanged.emit({
      SourceList: [...this.sourceList.itemsList],
      TargetList: [...this.targetList.itemsList]
    });
  }

  private reorderItems(up?: boolean) {
    if (!this.targetList.itemsList || this.targetList.itemsList.length < 2) {
      return;
    }

    let selectedIndices = [];
    this.targetList.itemsList.map((item, index) => {
      if (item.Selected) {
        selectedIndices.push(index);
      }
    });

    if (selectedIndices.length <= 0) {
      return;
    }
    if (
      (selectedIndices[0] === 0 && up) ||
      (selectedIndices[selectedIndices.length - 1] === this.targetList.itemsList.length - 1 && !up)
    ) {
      return;
    }
    if (!up) {
      selectedIndices = selectedIndices.reverse();
    }
    selectedIndices.forEach(index => this.reorderItem(index, up));

    this.listChanged.emit({ SourceList: [...this.sourceList.itemsList], TargetList: [...this.targetList.itemsList] });
  }

  private reorderItem(selectedIndex, up?) {
    if (up) {
      if (selectedIndex === 0) {
        return;
      }
      const prevItem = this.targetList.itemsList[selectedIndex - 1];
      this.targetList.itemsList[selectedIndex - 1] = this.targetList.itemsList[selectedIndex];
      this.targetList.itemsList[selectedIndex] = prevItem;
    } else {
      if (selectedIndex === this.targetList.itemsList.length - 1) {
        return;
      }
      const nextItem = this.targetList.itemsList[selectedIndex + 1];
      this.targetList.itemsList[selectedIndex + 1] = this.targetList.itemsList[selectedIndex];
      this.targetList.itemsList[selectedIndex] = nextItem;
    }
  }

  private dragdropItem(event: CdkDragDrop<IListItem[]>, droppedOnSource?: boolean) {
    if (event.container === event.previousContainer) {
      return;
    }
    const draglist = droppedOnSource ? this.targetList : this.sourceList;
    const droplist = droppedOnSource ? this.sourceList : this.targetList;
    const item = draglist.itemsList.splice(event.previousIndex, 1)[0];
    if (item.Selected && !droplist.allowMultiSelect) {
      droplist.itemsList.forEach(i => (i.Selected = false));
    }
    if (droplist.itemsList && droplist.itemsList.length) {
      droplist.itemsList.splice(event.currentIndex, 0, item);
    } else {
      droplist.itemsList = [item];
    }
    this.dropped.emit({ ...event, droppedOnSource });
    this.listChanged.emit({ SourceList: [...this.sourceList.itemsList], TargetList: [...this.targetList.itemsList] });
  }

  ngOnDestroy(): void {
    if (this.sub) {
      this.sub.unsubscribe();
      this.sub = null;
    }
  }
}
