import { ClientInfo } from '@clearsight-apps/models/common/client-info';

import {
  LicensedAppModel,
  NAV_APP,
  ON_DELETE_FAVORITE
} from '@clearsight-apps/shared/framework/collaborative-platform';
import { RecentFolderItem } from '@clearsight-apps/models/recent-items/recent-item';

export const ActivateAppEventName = 'ACTIVATE_APP';
export const OpenRecentItemEventName = 'OPEN_RECENT_ITEM';
export const RemoveRecentItemEventName = 'REMOVE_RECENT_ITEM';
export const CloseRecentItemEventName = 'CLOSE_RECENT_ITEM';
export const SetRecentItemsEventName = 'SET_RECENT_ITEMS';
export const NavClicked = 'NAV_CLICKED';
export const CloseAppEvent = 'CLOSE_APP';
export const RequestLicensedAppsEvent = 'LICENSED_APPS';
export const ReturnLicensedAppsEvent = 'RETURN_LICENSED_APPS';
export const RequestClientInfoEvent = 'CLIENT_INFO';
export const PostClientInfoEvent = 'POST_CLIENT_INFO';
export const OnClosedEventName = 'ON_CLOSED';

export interface CloseAppMessage {
  url: string;
}

export interface ActivateAppMessage {
  appName: string;
  relativeUrl?: string;
}

export interface RequestLicensedAppsMessage {
  appName: string;
}

export interface WindowEvent {
  appName?: string;
  toEventData(): any;
}

export class AppNavigation implements WindowEvent {
  constructor(public appName: string, private state: string) {}

  toEventData(): any {
    return { eventName: NAV_APP, data: { appName: this.appName, navTo: this.state } };
  }
}

export class CloseRecentItemEvent implements WindowEvent {
  constructor(public appName: string, private id: string, private featureKey: string) {}

  toEventData() {
    return {
      eventName: CloseRecentItemEventName,
      data: {
        appName: this.appName,
        data: {
          id: this.id,
          featureKey: this.featureKey
        }
      }
    };
  }
}

export class OpenRecentItemEvent implements WindowEvent {
  constructor(public appName: string, public item: RecentFolderItem) {}

  toEventData() {
    return { eventName: OpenRecentItemEventName, data: this.item };
  }
}

export class SetRecentItemsEvent implements WindowEvent {
  constructor(public appName: string, public openRecords: RecentFolderItem[]) {}

  toEventData() {
    return { eventName: SetRecentItemsEventName, openRecords: this.openRecords };
  }
}

export class RemoveRecentItemEvent implements WindowEvent {
  constructor(public appName: string, public id: string, public featureKey: string) {}

  toEventData() {
    return {
      eventName: OpenRecentItemEventName,
      data: {
        appName: this.appName,
        data: { id: this.id, featureKey: this.featureKey }
      }
    };
  }
}

export class PostLicensedApps implements WindowEvent {
  constructor(public appName: string, public apps: LicensedAppModel[]) {}

  toEventData() {
    return { eventName: ReturnLicensedAppsEvent, data: { appName: this.appName, licensedApps: this.apps } };
  }
}

export class PostClientInfo implements WindowEvent {
  constructor(public client: ClientInfo) {}

  toEventData() {
    return { eventName: PostClientInfoEvent, data: { clientInfo: this.client } };
  }
}

export class ToggleMobileMenu implements WindowEvent {
  constructor(public appName: string) {}

  toEventData() {
    return { eventName: NavClicked, data: { appName: this.appName } };
  }
}

export class OnClosedEvent implements WindowEvent {
  constructor() {}

  toEventData() {
    return { eventName: OnClosedEventName };
  }
}

export class OnDeleteFavoritesEvent implements WindowEvent {
  constructor(public appName: string, public viewId: any) {}

  toEventData() {
    return { eventName: ON_DELETE_FAVORITE, data: { appName: this.appName, ViewId: this.viewId } };
  }
}
