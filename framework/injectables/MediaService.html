import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';
import { hasValue } from '@clearsight-apps/common-functions/has-value';


export enum MediaType {
  Phone,
  Tablet,
  Desktop,
  Unknown
}

@Injectable({
  providedIn: 'root'
})
export class MediaService {
  private isIe: boolean;
  private isEdge: boolean;
  private isFireFox: boolean;
  private isSafari: boolean;
  private isCompability: boolean;
  private phoneQuery: MediaQueryList;
  private tabletQuery: MediaQueryList;
  private mobileQuery: MediaQueryList;
  private desktopQuery: MediaQueryList;
  public isMobileChanged$: BehaviorSubject<boolean>;
  public mediaTypeChanged = new BehaviorSubject<MediaType>(MediaType.Unknown);

  constructor() {
    this.wireEvents();
  }

  private wireEvents() {
    this.phoneQuery = window.matchMedia('(max-width: 767px)');

    this.tabletQuery = window.matchMedia('(min-width: 768px) and (max-width: 991px)');

    this.mobileQuery = window.matchMedia('(max-width: 991px)');
    this.isMobileChanged$ = new BehaviorSubject<boolean>(this.mobileQuery.matches);

    this.desktopQuery = window.matchMedia('(min-width: 992px)');

    this.mobileQuery.addListener((mql: MediaQueryListEvent) => {
      this.isMobileChanged$.next(mql.matches);
    });

    this.phoneQuery.addListener((mql: MediaQueryListEvent) => {
      if (mql.matches) {
        this.mediaTypeChanged.next(MediaType.Phone);
      }
    });

    this.tabletQuery.addListener((mql: MediaQueryListEvent) => {
      if (mql.matches) {
        this.mediaTypeChanged.next(MediaType.Tablet);
      }
    });

    this.desktopQuery.addListener((mql: MediaQueryListEvent) => {
      if (mql.matches) {
        this.mediaTypeChanged.next(MediaType.Desktop);
      }
    });

    if (this.mobileQuery.matches) {
      if (this.phoneQuery.matches) {
        this.mediaTypeChanged.next(MediaType.Phone);
      } else {
        this.mediaTypeChanged.next(MediaType.Tablet);
      }
    }
  }

  isIE(): boolean {
    if (hasValue(this.isIe)) {
      return this.isIe;
    }
    this.isIe = navigator.userAgent.indexOf('MSIE') !== -1 || navigator.appVersion.indexOf('Trident/') > 0;
    return this.isIe;
  }

  isEDGE(): boolean {
    if (hasValue(this.isEdge)) {
      return this.isEdge;
    }
    this.isEdge = !!navigator.userAgent.match(/Trident\/7\./) || window.navigator.userAgent.indexOf('Edge') > -1;
    return this.isEdge;
  }

  browserCompatibility(): boolean {
    if (hasValue(this.isCompability)) {
      return this.isCompability;
    }
    this.isIe = navigator.userAgent.indexOf('MSIE') !== -1 || navigator.appVersion.indexOf('Trident/') > 0;
    this.isEdge = !!navigator.userAgent.match(/Trident\/7\./) || window.navigator.userAgent.indexOf('Edge') > -1;
    this.isFireFox = window.navigator.userAgent.indexOf('Firefox') > -1;
    this.isSafari = /^((?!chrome|android).)*safari/i.test(navigator.userAgent);
    switch (true) {
      case this.isIe: {
        this.isCompability = this.isIe;
        break;
      }
      case this.isEdge: {
        this.isCompability = this.isEdge;
        break;
      }
      case this.isFireFox: {
        this.isCompability = this.isFireFox;
        break;
      }
      case this.isSafari: {
        this.isCompability = this.isSafari;
        break;
      }
    }
    return this.isCompability;
  }
}
