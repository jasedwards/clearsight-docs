import { Inject, Injectable, LOCALE_ID } from '@angular/core';
import {
  getCurrencySymbol,
  getLocaleNumberFormat,
  getLocaleNumberSymbol,
  NumberFormatStyle,
  NumberSymbol
} from '@angular/common';
import BigNumber from 'BigNumber.js';
import isUndefined from 'lodash-es/isUndefined';
import isEmpty from 'lodash-es/isEmpty';
import padEnd from 'lodash-es/padEnd';
import { NumberPattern } from './number-pattern.interface';
import { NumberConstants } from './number-constants';

@Injectable()
export class NumberFormatterService {
  static MAX_DIGITS = 22;
  static DECIMAL_SEP = '.';
  static ZERO_CHAR = '0';
  static PATTERN_SEP = ';';
  static GROUP_SEP = ',';
  static DIGIT_CHAR = '#';
  static CURRENCY_CHAR = '¤';
  static PERCENT_CHAR = '%';

  GroupSeparator: string;
  DecimalSeparator: string;

  constructor(@Inject(LOCALE_ID) private locale: string) {
    this.initialize();
  }

  static replaceSymbol(val: string, style: NumberFormatStyle, locale: string, showSymbol = false): string {
    switch (style) {
      case NumberFormatStyle.Currency:
        if (showSymbol) {
          const symbol = getCurrencySymbol('USD', 'narrow');
          return val
            .replace(NumberFormatterService.CURRENCY_CHAR, symbol)
            .replace(NumberFormatterService.CURRENCY_CHAR, '');
        } else {
          return val.replace(NumberFormatterService.CURRENCY_CHAR, '');
        }
      case NumberFormatStyle.Percent:
        const perc = showSymbol ? getLocaleNumberSymbol(locale, NumberSymbol.PercentSign) : '';
        return val.replace(NumberFormatterService.PERCENT_CHAR, perc);
      default:
        return val;
    }
  }

  static parseNumberFormat(format: string, minusSign = '-'): NumberPattern {
    const p = {
      minInt: 1,
      minFrac: 0,
      maxFrac: 0,
      posPre: '',
      posSuf: '',
      negPre: '',
      negSuf: '',
      gSize: 0,
      lgSize: 0
    };

    const patternParts = format.split(NumberConstants.PATTERN_SEP);
    const positive = patternParts[0];
    const negative = patternParts[1];

    const positiveParts =
        positive.indexOf(NumberConstants.DECIMAL_SEP) !== -1
          ? positive.split(NumberConstants.DECIMAL_SEP)
          : [
              positive.substring(0, positive.lastIndexOf(NumberConstants.ZERO_CHAR) + 1),
              positive.substring(positive.lastIndexOf(NumberConstants.ZERO_CHAR) + 1)
            ],
      integer = positiveParts[0],
      fraction = positiveParts[1] || '';

    p.posPre = integer.substr(0, integer.indexOf(NumberConstants.DIGIT_CHAR));

    for (let i = 0; i < fraction.length; i++) {
      const ch = fraction.charAt(i);
      if (ch === NumberConstants.ZERO_CHAR) {
        p.minFrac = p.maxFrac = i + 1;
      } else if (ch === NumberConstants.DIGIT_CHAR) {
        p.maxFrac = i + 1;
      } else {
        p.posSuf += ch;
      }
    }

    const groups = integer.split(NumberConstants.GROUP_SEP);
    p.gSize = groups[1] ? groups[1].length : 0;
    p.lgSize = groups[2] || groups[1] ? (groups[2] || groups[1]).length : 0;

    if (negative) {
      const trunkLen = positive.length - p.posPre.length - p.posSuf.length,
        pos = negative.indexOf(NumberConstants.DIGIT_CHAR);

      p.negPre = negative.substr(0, pos).replace(/'/g, '');
      p.negSuf = negative.substr(pos + trunkLen).replace(/'/g, '');
    } else {
      p.negPre = minusSign + p.posPre;
      p.negSuf = p.posSuf;
    }

    return p;
  }

  private initialize() {
    BigNumber.config({ ERRORS: false });
    this.DecimalSeparator = getLocaleNumberSymbol(this.locale, NumberSymbol.CurrencyDecimal);
    this.GroupSeparator = getLocaleNumberSymbol(this.locale, NumberSymbol.CurrencyGroup);
  }

  format({
    value,
    fractionSize,
    style = NumberFormatStyle.Decimal,
    showSymbol = false,
    fillDecimals = false,
    round = false
  }: {
    value: number | string;
    fractionSize: number;
    style?: NumberFormatStyle;
    showSymbol?: boolean;
    fillDecimals?: boolean;
    round?: boolean;
  }): string {
    const format = getLocaleNumberFormat(this.locale, style);
    let num;

    if (typeof value === 'string' && !isNaN(+value - parseFloat(value))) {
      num = +value;
    } else if (typeof value !== 'number') {
      // res.error = `${value} is not a number`;
      // return res;
      return value;
    } else {
      num = value;
    }

    if (style === NumberFormatStyle.Percent) {
      num = num * 100;
    }

    const isNegative = num < 0;
    const pattern = NumberFormatterService.parseNumberFormat(
      format,
      getLocaleNumberSymbol(this.locale, NumberSymbol.MinusSign)
    );
    num = new BigNumber(num).abs();
    const numStr = num + '';

    let formattedText = '';
    const parts = [];
    let hasExponent = false;
    if (numStr.indexOf('e') !== -1) {
      const match = numStr.match(NumberConstants._EXPONENT_REGEXP);
      if (match && match[2] === '-' && +match[3] > fractionSize + 1) {
        num = 0;
      } else {
        formattedText = numStr;
        hasExponent = true;
      }
    }

    if (!hasExponent) {
      const fractionLen = (numStr.split('.')[1] || '').length;
      if (isUndefined(fractionSize)) {
        fractionSize = Math.min(Math.max(pattern.minFrac, fractionLen), pattern.maxFrac);
      }

      if (round) {
        num = +(Math.round(+(num.toString() + 'e' + fractionSize)).toString() + 'e' + -fractionSize);
      }

      const fractionArr = ('' + num).split('.');
      const whole = fractionArr[0];
      const fraction = fractionArr[1] || '';
      let i,
        pos = 0,
        lgroup,
        group;
      if (NumberConstants.NegativeSuffixCultures.indexOf(this.locale) > -1) {
        lgroup = pattern.maxFrac;
        group = pattern.maxFrac;
      } else {
        lgroup = pattern.lgSize;
        group = pattern.gSize;
      }

      if (whole.length >= lgroup + group) {
        pos = whole.length - lgroup;
        for (i = 0; i < pos; i++) {
          if ((pos - i) % group === 0 && i !== 0) {
            formattedText += this.GroupSeparator;
          }
          formattedText += whole.charAt(i);
        }
      }

      for (i = pos; i < whole.length; i++) {
        if ((whole.length - i) % lgroup === 0 && i !== 0) {
          formattedText += this.GroupSeparator;
        }
        formattedText += whole.charAt(i);
      }
      if (fractionSize && fractionSize !== 0 && !isEmpty(fraction)) {
        formattedText += this.DecimalSeparator + fraction.substr(0, fractionSize);
      }
    } else {
      if (fractionSize > 0 && num < 1) {
        formattedText = (num as number).toFixed(fractionSize);
        num = parseFloat(formattedText);
      }
    }

    if (fillDecimals) {
      const decLen = formattedText.split(this.DecimalSeparator);
      if (decLen) {
        if (decLen.length < 2) {
          decLen.push('');
        }
        if (fractionSize > decLen[1].length) {
          decLen[1] = padEnd(decLen[1], fractionSize, '0');
          formattedText = decLen.join(this.DecimalSeparator);
        }
      }
    }

    if (NumberConstants.NegativeSuffixCultures.indexOf(this.locale) > -1) {
      parts.push(
        isNegative ? pattern.negSuf : pattern.posPre,
        formattedText,
        isNegative ? pattern.negPre : pattern.posSuf
      );
    } else {
      parts.push(
        isNegative ? pattern.negPre : pattern.posPre,
        formattedText,
        isNegative ? pattern.negSuf : pattern.posSuf
      );
    }
    return NumberFormatterService.replaceSymbol(parts.join(''), style, this.locale, showSymbol);
  }
}
